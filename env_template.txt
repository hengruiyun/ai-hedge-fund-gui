# AI对冲基金 - 环境变量配置模板
# 请将此文件重命名为 .env 并填入您的API密钥

# ===== LLM API密钥 (至少需要设置一个) =====

# OpenAI API密钥 - 用于GPT-4等模型
# 获取地址: https://platform.openai.com/api-keys
OPENAI_API_KEY=your_openai_api_key_here

# Groq API密钥 - 高速推理服务
# 获取地址: https://console.groq.com/keys
GROQ_API_KEY=your_groq_api_key_here

# Anthropic API密钥 - 用于Claude模型
# 获取地址: https://console.anthropic.com/
ANTHROPIC_API_KEY=your_anthropic_api_key_here

# DeepSeek API密钥 - 中国AI模型
# 获取地址: https://platform.deepseek.com/api_keys
DEEPSEEK_API_KEY=your_deepseek_api_key_here

# Google API密钥 - 用于Gemini模型
# 获取地址: https://makersuite.google.com/app/apikey
GOOGLE_API_KEY=your_google_api_key_here

# ===== 金融数据API密钥 (可选) =====

# Financial Datasets API密钥
# 获取地址: https://financialdatasets.ai/
# 注意: AAPL, GOOGL, MSFT, NVDA, TSLA 无需此密钥
FINANCIAL_DATASETS_API_KEY=your_financial_datasets_api_key_here
